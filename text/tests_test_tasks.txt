import pytest


@pytest.mark.asyncio
async def test_patch_task_status_only_changes_status(client, create_user):
    create_user("task-owner@example.com")
    login = await client.post(
        "/auth/login",
        json={"email": "task-owner@example.com", "password": "password123"},
    )
    project = await client.post(
        "/projects",
        json={"title": "Task Project", "description": None, "status": "active"},
        headers={"Authorization": f"Bearer {login.json()['access_token']}"},
    )
    task = await client.post(
        f"/projects/{project.json()['id']}/tasks",
        json={"title": "Task 1", "description": "Keep", "status": "todo", "priority": "high"},
        headers={"Authorization": f"Bearer {login.json()['access_token']}"},
    )

    response = await client.patch(
        f"/tasks/{task.json()['id']}/status",
        json={"status": "done"},
        headers={"Authorization": f"Bearer {login.json()['access_token']}"},
    )
    assert response.status_code == 200
    assert response.json()["status"] == "done"
    assert response.json()["description"] == "Keep"

@pytest.mark.asyncio
async def test_attach_and_detach_tag(client, create_user):
    create_user("tag-owner@example.com")
    login = await client.post(
        "/auth/login",
        json={"email": "tag-owner@example.com", "password": "password123"},
    )
    headers = {"Authorization": f"Bearer {login.json()['access_token']}"}
    project = await client.post("/projects", json={"title": "Tags", "description": None, "status": "active"}, headers=headers)
    task = await client.post(
        f"/projects/{project.json()['id']}/tasks",
        json={"title": "Tag me", "description": None, "status": "todo", "priority": "medium"},
        headers=headers,
    )
    tag = await client.post("/tags", json={"name": "backend", "colour": "#112233"}, headers=headers)

    attach = await client.post(f"/tasks/{task.json()['id']}/tags/{tag.json()['id']}", headers=headers)
    assert attach.status_code == 200

    detach = await client.delete(f"/tasks/{task.json()['id']}/tags/{tag.json()['id']}", headers=headers)
    assert detach.status_code == 204

    login = await client.post(
        "/auth/login",
        json={"email": "commenter@example.com", "password": "password123"},
    )
    headers = {"Authorization": f"Bearer {login.json()['access_token']}"}
    project = await client.post("/projects", json={"title": "Comments", "description": None, "status": "active"}, headers=headers)
    task = await client.post(

        headers=headers,
    )

@pytest.mark.asyncio
async def test_revoked_api_key_is_rejected(client, create_user):
    create_user("revoked-key@example.com")
    login = await client.post(
        "/auth/login",
        json={"email": "revoked-key@example.com", "password": "password123"},
    )
    jwt_headers = {"Authorization": f"Bearer {login.json()['access_token']}"}
    key = await client.post("/auth/api-keys", json={"name": "temp"}, headers=jwt_headers)
    await client.delete(f"/auth/api-keys/{key.json()['id']}", headers=jwt_headers)
    response = await client.get("/auth/me", headers={"X-API-Key": key.json()["key"]})
    assert response.status_code == 401
